# ------------------------------------------------------------
# Opcion A: ejecutar con Ollama local
# ------------------------------------------------------------

# En una terminal, levantar Ollama si no esta corriendo
ollama serve

# En otra terminal, validar que el servidor responde
curl http://localhost:11434/api/tags

# Descargar modelo, si no esta descargado
ollama pull llama3.1

# Configurar provider para esta terminal
export OLLAMA_HOST="http://localhost:11434"
export OLLAMA_MODEL="llama3.1"

# Ejecutar agente con calculadora
python -m mia_agents.cli run --module student_framework \
  --message "¿Cuánto es 17 * 23? Usá la calculadora."

# Ejecutar agente con contador de palabras
python -m mia_agents.cli run --module student_framework \
  --message "¿Cuántas palabras tiene la frase 'el agente usa herramientas para resolver tareas'?"

# Ejecutar agente con lector de archivos.
# El archivo debe existir dentro de sample_files/.
python -m mia_agents.cli run --module student_framework \
  --message "Leé el archivo hola_mundo.md y decime qué dice."

# Ejecutar agente combinando file_reader + word_counter.
